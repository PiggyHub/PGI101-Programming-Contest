## Example 2 Sort
```c++
#include<iostream>
#include<algorithm>
using namespace std;

bool cmp(int a, int b){
	return a>b;
}

int main(){
	int a[10]={1,5,2,3,4};
	sort(a,a+5,cmp);
	for(int i=0; i<=4; i++) cout<<a[i]<<" ";
	return 0;
}
```
Output
```
5 4 3 2 1
```
